GreenGo

Arduino

GreenGo é um jogo interativo de teste de reflexos, ideal para desafiar a rapidez e precisão dos jogadores. O projeto consiste em uma caixa equipada com luzes LED um botão e um buzzer, que juntos criam uma experiência divertida e envolvente. O objetivo é simples: os jogadores devem o botão o mais rápido possível assim que a luz verde acender.

Arduino  

O Arduino é uma placa de prototipagem eletrônica de código aberto e hardware livre, composto por um microcontrolador Atmel, circuitos de entrada e saída e programação via IDE. Seu software é desenvolvido por meio de linguagem baseada em C/C++, usando um ambiente gráfico escrito em Java. Sendo assim, a programação do Arduino dispensa equipamentos extras além de um cabo USB, permitindo infinitas modificações, conforme a necessidade de cada usuário.


Funcionamento  

Para começar, o jogador deve pressionar o botão. Após isso, o LED vermelho acenderá, indicando que ele deve esperar o próximo sinal. Com o LED vermelho ligado, o jogador deverá aguardar o momento em que o led verde irá acender e pressionar o botão nesse instante. Se o jogador acerta e pressiona o botão rapidamente, o LED verde pisca e um som positivo é emitido pelo buzzer, indicando o sucesso. Caso o jogador aperte o botão antes, ou depois, do LED verde o buzzer vai emitir um som de erro e o jogo recomeçará.


Lista de Componentes

Componentes usados : Buzzer - O buzzer é um pequeno dispositivo que gera som, geralmente em forma de bipes (beeps). Ele pode ser ativo (emite som contínuo quando energizado) ou passivo (precisa de sinais PWM para gerar sons variados). Botão - Um botão é um componente eletrônico simples que, ao ser pressionado, permite a passagem de corrente elétrica (ou seja, fecha um circuito). Quando solto, o circuito é interrompido. LED - Um LED é um diodo emissor de luz que acende ao receber corrente elétrica na polaridade correta. É muito usado em projetos para indicar estados visuais de forma simples. LCD - Um display de cristal líquido usado para exibir textos, números ou outras informações visuais. O mais comum em projetos com Arduino é o LCD 16x2 (16 caracteres por linha, 2 linhas).

#include
#include // Biblioteca para o LCD I2C
// Inicializa o display LCD I2C com endereço 0x27 e tamanho 16x2
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Define os pinos dos LEDs, botão e buzzer
const int ledVermelho = 2;
const int ledVerde = 4;
const int botao = 7;
const int buzzer = 8;
// Pino para o buzzer

// Variáveis para armazenar o estado do jogo
bool botaoPressionado = false;
bool jogoEmAndamento = false;

// Notas musicais (frequências) para a música de vitória (Mario) e de derrota (Pac-Man)
#define NOTE_C4 262
#define NOTE_E4 330
#define NOTE_G4 392
#define NOTE_C5 523
#define NOTE_G3 196
#define NOTE_F4 349
#define NOTE_A3 220
#define NOTE_B3 247

void setup() {
// Configura os pinos como saída ou entrada
pinMode(ledVermelho, OUTPUT);
pinMode(ledVerde, OUTPUT);
pinMode(botao, INPUT_PULLUP);
pinMode(buzzer, OUTPUT);

// Inicia a comunicação serial para feedback
Serial.begin(9600);

// Inicia o LCD e exibe o nome do jogo
lcd.init(); // Inicializa o LCD
lcd.backlight(); // Liga a luz de fundo
mostrarInicio();
}

void loop() {
// Aguarda o jogador apertar o botão para iniciar o jogo
Serial.println("Pressione o botão para iniciar o jogo...");
lcd.clear();
lcd.setCursor(0, 0);
lcd.print("Aperte o botao");

while (digitalRead(botao) == HIGH) {
// Espera até que o botão seja pressionado (botão LOW)
}

// Uma vez que o botão é pressionado, o jogo começa
delay(500); // Pequeno atraso para evitar leituras acidentais
Serial.println("Jogo iniciado! Espere o LED verde.");
digitalWrite(ledVerde, LOW); // Garante que o LED verde esteja apagado ao iniciar uma nova rodada
lcd.clear();
lcd.setCursor(0, 0);
lcd.print("Olhe o LED verde");

// Gera um tempo aleatório para o LED vermelho
int tempoVermelho = random(500, 2000); // Tempo aleatório entre 0,5 e 2 segundos

// Liga o LED vermelho para indicar que o jogador deve esperar
digitalWrite(ledVermelho, HIGH);
delay(tempoVermelho);
digitalWrite(ledVermelho, LOW);

// Liga o LED verde e aguarda o jogador apertar o botão
digitalWrite(ledVerde, HIGH); // LED verde acende para o jogador apertar o botão
jogoEmAndamento = true;

// Loop para verificar se o jogador aperta o botão a tempo
unsigned long startTime = millis();
while (jogoEmAndamento) {
// Verifica se o botão foi pressionado
if (digitalRead(botao) == LOW) {
botaoPressionado = true;
jogoEmAndamento = false; // Termina o loop
}

// Tempo limite para o jogador apertar o botão (por exemplo, 2 segundos)
if (millis() - startTime > 2000) {
jogoEmAndamento = false; // Tempo esgotado
}
}

// Avalia o resultado
if (botaoPressionado) {
// Jogador apertou o botão no tempo correto
Serial.println("Parabéns! Você apertou no tempo certo!");
lcd.clear();
lcd.setCursor(0, 0);
lcd.print("Você663 Venceu!");
tocarMelodiaMario();
delay(500); // Tempo curto em que o LED verde permanece aceso após o acerto
} else {
// Jogador não apertou o botão a tempo
Serial.println("Tempo esgotado ou botão apertado fora de hora!");
lcd.clear();
lcd.setCursor(0, 0);
lcd.print("Derrota! Tente");
lcd.setCursor(0, 1);
lcd.print("novamente.");
tocarMelodiaPacman();
// Pisca o LED vermelho para indicar erro
piscarLed(ledVermelho, 3, 300);
}

// Desliga o LED verde antes de iniciar a próxima rodada
digitalWrite(ledVerde, LOW);
botaoPressionado = false;
delay(1000); // Pausa antes de começar uma nova rodada
}
// Função para exibir o nome do jogo no início
void mostrarInicio() {
lcd.clear();
lcd.setCursor(0, 0);
lcd.print("Bem-vindo ao");
lcd.setCursor(0, 1);
lcd.print("GREENGO!");
delay(2000); // Exibe a mensagem por 2 segundos antes de iniciar o jogo
}

// Função para piscar o LED
void piscarLed(int pino, int vezes, int intervalo) {
for (int i = 0; i < vezes; i++) {
digitalWrite(pino, HIGH);
delay(intervalo);
digitalWrite(pino, LOW);
delay(intervalo);
}
}

// Função para tocar a melodia de vitória (tema Mario Bros simplificado)
void tocarMelodiaMario() {
int melody[] = {NOTE_E4, NOTE_E4, 0, NOTE_E4, 0, NOTE_C4, NOTE_E4, 0, NOTE_G4, 0, 0, 0, NOTE_G3, 0, 0, 0};
int noteDurations[] = {200, 200, 100, 200, 100, 200, 200, 100, 200, 300, 100, 300, 200, 100, 300, 400};

for (int i = 0; i < 16; i++) {
int noteDuration = noteDurations[i];
if (melody[i] == 0) {
delay(noteDuration); // Pausa para as notas 0 (silêncio)
} else {
tone(buzzer, melody[i], noteDuration);
}
delay(noteDuration * 1.3); // Pequeno atraso entre as notas
}
noTone(buzzer);
}

// Função para tocar a melodia de derrota (morte Pac-Man)
void tocarMelodiaPacman() {
int melody[] = {NOTE_C4, NOTE_B3, NOTE_A3, NOTE_G3, NOTE_F4, NOTE_F4};
int noteDurations[] = {100, 100, 100, 100, 300, 300};

for (int i = 0; i < 6; i++) {
int noteDuration = noteDurations[i];
tone(buzzer, melody[i], noteDuration);
delay(noteDuration * 1.3); // Pequeno atraso entre as notas
}
noTone(buzzer);
}
